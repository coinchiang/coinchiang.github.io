/**
* 如果去掉下面方法中的default则会报错
* 对于一个有返回值的方法，内部流程控制必须确保逻辑上的完备性，无论何种情况都应该返回一个值
*/
public static double calc(char mode, double a, double b) {
    switch (mode) {
        case '+':	
            return a + b; 
        case '-': 	
            return a - b; 
        case '*': 	
            return a * b;
        case '/':	
            return a / b;
        default :  
            return -10000;//会自动提升，没有警告
    }
}
